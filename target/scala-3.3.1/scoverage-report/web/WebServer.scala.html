108   @cask.postJson("/evaluate1")
109   def evaluate1(argast: ujson.Value): ujson.Obj =
110     // Function to try to evaluate the expression.
111     def tryeval1(argast: ujson.Value): Try[ujson.Obj] =
112       Try {
113         // Parse the JSON value into an Either.
114         val eitherExpression: Either[List[String], Expression[Int]] =
115           read[Either[List[String], Expression[Int]]](argast)
116         eitherExpression match
117           case Right(expression) =>
118             // Evaluate the expression.
119             val intResultAsExpression = Expression
120               .evaluateValidExpression(
121                 Expression.validateExpression(expression)
122               )
123               .flatMap(x => Expression.validateExpression(Num(x)))
124             // Convert the result to JSON.
125             val intResultJson: String = write(intResultAsExpression)
126             intResultAsExpression match
127               case Right(_) =>
128                 val result = Expression
129                   .evaluateValidExpression(intResultAsExpression)
130                   .right
131                   .get
132                 // Store the result in the ResultsHolder.
133                 ResultsHolder.shareintExpressionEvaluated = Right(Some(result))
134               case Left(_) =>
135                 ResultsHolder.shareintExpressionEvaluated = Right(None)
136             // Return the result as a JSON object.
137             ujson.Obj("result" -> intResultJson)
138           case Left(error) =>
139             val dfErrorJson: String = write(eitherExpression)
140             // Return the error as a JSON object.
141             ujson.Obj("error" -> dfErrorJson)
142       }
143     end tryeval1
144     val result: Try[ujson.Obj] = tryeval1(argast)
145     result match
146       case Success(value)     => value
147       case Failure(exception) =>
148         // Return the exception message as a JSON object.
