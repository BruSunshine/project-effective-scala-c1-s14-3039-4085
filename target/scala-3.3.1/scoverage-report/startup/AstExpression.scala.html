29   /** Evaluates an arithmetic expression.
30     *
31     * @param expression
32     *   The expression to evaluate.
33     * @param ops
34     *   The given instance of `ArithmeticOperation` for the type `T`.
35     * @return
36     *   The result of the evaluation.
37     */
38   def evaluateValidExpression[T: OperationValidator](
39       expression: Either[List[String], Expression[T]]
40   )(using
41       ops: ArithmeticOperation[T]
42   ): Either[List[String], T] =
43     try
44       expression match
45         case Right(exp) =>
46           exp match
47             case Num[T](num) => Right(num)
48             case Mult[T](a, b) =>
49               for
50                 A <- evaluateValidExpression(Right(a))
51                 B <- evaluateValidExpression(Right(b))
52               yield
53                 if summon[OperationValidator[T]].validate(A, B, "mul") then
54                   ops.mul(A, B)
55                 else throw new Exception("Invalid mul operation")
56             case Plus[T](a, b) =>
57               for
58                 A <- evaluateValidExpression(Right(a))
59                 B <- evaluateValidExpression(Right(b))
60               yield
61                 if summon[OperationValidator[T]].validate(A, B, "add") then
62                   ops.add(A, B)
63                 else throw new Exception("Invalid plus operation")
64         case Left(errors) => Left(errors)
65     catch case error => Left(List(s"$error"))
66   end evaluateValidExpression

68   /** Provides an implicit `ReadWriter[Expression[T]]` instance for any type `T`
69     * that has a `ReadWriter[T]` instance.
70     *
71     * This instance can read and write `Expression[T]` instances as JSON. It's
72     * created by merging the `ReadWriter` instances for `Num[T]`, `Plus[T]`, and
73     * `Mult[T]`.
74     *
75     * @param RW[T]
76     *   The given `ReadWriter[T]` instance for the type `T`.
77     * @return
78     *   The `ReadWriter[Expression[T]]` instance.
79     */

87   /** Validates an arithmetic expression.
88     *
89     * @param expression
90     *   The expression to validate.
91     * @return
92     *   The result of the validation.
93     */
94   def validateExpression[T: ArgumentValidator](
95       expression: Expression[T]
96   ): Either[List[String], Expression[T]] =
97     def valExpIntern[T: ArgumentValidator](
98         expression: Expression[T],
99         errorsAccumulator: List[String]
100     ): Either[List[String], Expression[T]] =
101       try
102         expression match
103           case Num(n) =>
104             if summon[ArgumentValidator[T]].validate(n) then Right(expression)
105             else Left(errorsAccumulator :+ s"Invalid Num argument $n")
106           case Mult(a, b) =>
107             (
108               valExpIntern(a, errorsAccumulator),
109               valExpIntern(b, errorsAccumulator)
110             ) match
111               case (Right(_), Right(_)) => Right(expression)
112               case (Left(error), Right(_)) =>
113                 Left(errorsAccumulator ++ error)
114               case (Right(_), Left(error)) =>
115                 Left(errorsAccumulator ++ error)
116               case (Left(error1), Left(error2)) =>
117                 Left(
118                   errorsAccumulator ++ error1 ++ error2
119                 )
120           case Plus(a, b) =>
121             (
122               valExpIntern(a, errorsAccumulator),
123               valExpIntern(b, errorsAccumulator)
124             ) match
125               case (Right(_), Right(_)) => Right(expression)
126               case (Left(error), Right(_)) =>
127                 Left(errorsAccumulator ++ error)
128               case (Right(_), Left(error)) =>
129                 Left(errorsAccumulator ++ error)
130               case (Left(error1), Left(error2)) =>
131                 Left(
132                   errorsAccumulator ++ error1 ++ error2
133                 )
134           case null => Left(errorsAccumulator ++ List(s"null expression"))
135       catch
136         case error =>
137           Left(errorsAccumulator :+ s"error in expression validation: $error")
138     end valExpIntern
139     valExpIntern(expression, List.empty[String])
140   end validateExpression
